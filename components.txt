                        this.setState({count:this.state.count+1})

                    }
                    decrement=()=>{
                        this.setState({count:this.state.count-1})
                    }
          
        render(){
            return(
                <div>
                    <h1>My component</h1>
                    <h1>{this.state.count}</h1>
                    <button onClick={this.increment}>increment</button>
                    <button onClick={this.decrement}>decrement</button>
